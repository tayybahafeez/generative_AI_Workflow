# Use an official Python runtime as a parent image
FROM python:3.9-slim

# Set the working directory in the container
WORKDIR /app

# Copy the current directory contents into the container at /app
COPY . /app

# Install any needed dependencies specified in requirements.txt
RUN pip install --no-cache-dir -r requirements.txt

# Install necessary libraries and FastAPI dependencies
RUN pip install --no-cache-dir fastapi uvicorn ollama

# Expose port 8000 for the FastAPI app to be accessible
EXPOSE 8000

# Define the command to run your FastAPI app using uvicorn
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
